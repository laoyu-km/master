### reduce 方法

```js
Array.prototype.myReduce = function(callback, iniValue) {
    if (typeof callback !== 'function') {
        throw new Error('callback must be a function');
    }

    var pre = iniValue,
        cur,
        len = this.length;

    for (var i = 0; i < len; i++) {
        cur = this[i];
        pre = callback(pre, cur, i, this);
    }

    return pre;

}
```
